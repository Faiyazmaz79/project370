1. Core Entities


CREATE TABLE Department (
    dept_id             VARCHAR(10)  PRIMARY KEY,
    dept_name           VARCHAR(100) NOT NULL,
    total_no_students   INT DEFAULT 0
);

CREATE TABLE Faculty (
    faculty_id      VARCHAR(10)  PRIMARY KEY,
    faculty_type    VARCHAR(30),
    specialization  VARCHAR(100),
    name            VARCHAR(100) NOT NULL,
    email           VARCHAR(100) NOT NULL UNIQUE,
    dept_id         VARCHAR(10),
    FOREIGN KEY (dept_id) REFERENCES Department(dept_id)
);

CREATE TABLE Student (
    Student_id      VARCHAR(10)  PRIMARY KEY,
    DOB             DATE,
    Email           VARCHAR(100) NOT NULL UNIQUE,
    Fname           VARCHAR(50)  NOT NULL,
    Lname           VARCHAR(50)  NOT NULL,
    cgpa            DECIMAL(4,2) DEFAULT 0.00,
    Student_type    VARCHAR(20)  DEFAULT 'Undergrad',
    Gender          VARCHAR(10),
    dept_id         VARCHAR(10),
    FOREIGN KEY (dept_id) REFERENCES Department(dept_id)
);

CREATE TABLE Student_Phone (
    Phone           VARCHAR(20),
    Student_id      VARCHAR(10),
    PRIMARY KEY (Phone, Student_id),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

2. Facilities (Optimized Classrooms)


CREATE TABLE Classroom (
    room_no         VARCHAR(20)  PRIMARY KEY,
    capacity        INT NOT NULL,
    projector       BOOLEAN DEFAULT FALSE,
    date            DATE,
    time            TIME,
    pcs             INT DEFAULT 0,
    Equipment_List  VARCHAR(255)       
);

CREATE TABLE Classroom_Lab_Room (
    lab_room        VARCHAR(50),
    room_no         VARCHAR(20),
    Occupied_Slots  INT DEFAULT 0,
    Free_Slots      INT DEFAULT 0,
    PRIMARY KEY (lab_room),
    FOREIGN KEY (room_no) REFERENCES Classroom(room_no)
);

CREATE TABLE Classroom_Theory_Room (
    theory_room     VARCHAR(50),
    room_no         VARCHAR(20),
    Occupied_Slots  INT DEFAULT 0,
    Free_Slots      INT DEFAULT 0,
    PRIMARY KEY (theory_room),
    FOREIGN KEY (room_no) REFERENCES Classroom(room_no)
);

3. Academics



CREATE TABLE Course (
    course_code         VARCHAR(10)  PRIMARY KEY,
    Semester_id         VARCHAR(10)  NOT NULL,
    credit_hours        INT NOT NULL DEFAULT 3,
    Enrollment_date     DATE,
    max_capacity        INT DEFAULT 40,
    Start_date          DATE,
    End_date            DATE,
    Max_credits         INT DEFAULT 15,
    dept_id             VARCHAR(10),
    room_no             VARCHAR(20),
    FOREIGN KEY (dept_id) REFERENCES Department(dept_id),
    FOREIGN KEY (room_no) REFERENCES Classroom(room_no)
);

CREATE TABLE Prerequisite (
    course_code         VARCHAR(10),
    prereq_course_code  VARCHAR(10),
    PRIMARY KEY (course_code, prereq_course_code),
    FOREIGN KEY (course_code)        REFERENCES Course(course_code),
    FOREIGN KEY (prereq_course_code) REFERENCES Course(course_code)
);

CREATE TABLE Enrolled_In (
    Student_id      VARCHAR(10),
    course_code     VARCHAR(10),
    PRIMARY KEY (Student_id, course_code),
    FOREIGN KEY (Student_id)  REFERENCES Student(Student_id),
    FOREIGN KEY (course_code) REFERENCES Course(course_code)
);

CREATE TABLE Grade (
    Student_id      VARCHAR(10),
    Course_code     VARCHAR(10),
    Grade_Point     VARCHAR(5),        
    Marks_obtained  DECIMAL(5,2),
    PRIMARY KEY (Student_id, Course_code),
    FOREIGN KEY (Student_id)  REFERENCES Student(Student_id),
    FOREIGN KEY (Course_code) REFERENCES Course(course_code)
);

CREATE TABLE Exam (
    Exam_id         INT AUTO_INCREMENT PRIMARY KEY,
    Exam_type       VARCHAR(20) NOT NULL,
    Room_number     VARCHAR(20),
    Start_time      TIME,
    Exam_date       DATE,
    Student_id      VARCHAR(10),
    FOREIGN KEY (Student_id)  REFERENCES Student(Student_id),
    FOREIGN KEY (Room_number) REFERENCES Classroom(room_no)
);

CREATE TABLE Section (
    section_no      VARCHAR(10)  PRIMARY KEY,
    room_no         VARCHAR(20),
    date            DATE,
    time            TIME,
    FOREIGN KEY (room_no) REFERENCES Classroom(room_no)
);

CREATE TABLE Section_Timing (
    Timing          VARCHAR(50),
    section_no      VARCHAR(10),
    Semester_id     VARCHAR(10),
    PRIMARY KEY (section_no, Semester_id),
    FOREIGN KEY (section_no) REFERENCES Section(section_no)
);

4. University Portal Features



CREATE TABLE Note (
    Title           VARCHAR(200) NOT NULL,
    Rating          DECIMAL(3,2) DEFAULT 0.00,
    File_Type       VARCHAR(20),
    Download        INT DEFAULT 0,
    Upload_Date     DATE,
    Student_id      VARCHAR(10),
    PRIMARY KEY (Title, Student_id),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

CREATE TABLE Reward_Points (
    Reward_id       INT AUTO_INCREMENT PRIMARY KEY,
    Rank            INT,
    Points_Awarded  INT DEFAULT 0,
    Student_id      VARCHAR(10),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

CREATE TABLE Clubs (
    Club_Name       VARCHAR(100) PRIMARY KEY,
    NO_of_members   INT DEFAULT 0,
    Activities      TEXT,
    Student_id      VARCHAR(10),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

CREATE TABLE Complaint (
    Complain_id     INT AUTO_INCREMENT PRIMARY KEY,
    Statement       TEXT,
    Issue_Type      VARCHAR(100),
    submitted_date  DATE,
    Student_id      VARCHAR(10),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

CREATE TABLE Files (
    Student_id      VARCHAR(10),
    Complain_id     INT,
    PRIMARY KEY (Student_id, Complain_id),
    FOREIGN KEY (Student_id)  REFERENCES Student(Student_id),
    FOREIGN KEY (Complain_id) REFERENCES Complaint(Complain_id)
);

CREATE TABLE AcademicRoutine (
    Student_id      VARCHAR(10),
    Courses         VARCHAR(100),
    Exam_date       DATE,
    Exam_routine    TEXT,
    Attribute       VARCHAR(100),
    PRIMARY KEY (Student_id, Courses),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

CREATE TABLE Semester_Fees (
    Payment_id      INT AUTO_INCREMENT PRIMARY KEY,
    Payment_date    DATE,
    Segment         VARCHAR(50),
    Amount          DECIMAL(10,2) NOT NULL,
    Student_id      VARCHAR(10),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);

CREATE TABLE AnalyticsDashboard (
    Course_Section      VARCHAR(50) PRIMARY KEY,
    Highest_Grade       VARCHAR(5),        
    Class_performance   VARCHAR(100),
    Lowest_Grade        VARCHAR(5)         
);

CREATE TABLE Smart_Recommendation (
    Student_id       VARCHAR(10),
    Courses_done     TEXT,
    courses_not_done TEXT,
    PRIMARY KEY (Student_id),
    FOREIGN KEY (Student_id) REFERENCES Student(Student_id)
);
